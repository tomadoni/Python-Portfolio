import pandas as pd
import numpy as np
from sklearn.metrics import r2_score, mean_absolute_error, mean_squared_error
import matplotlib.pyplot as plt
from statsmodels.nonparametric.smoothers_lowess import lowess


df = pd.read_excel('House_Prices_PRED.xlsx', sheet_name='PB2')

#calculate SSE and MSE
y_true = df['SalePrice']
y_pred = df['SalePrice_MP']
sse = np.sum((y_true - y_pred) ** 2)
mse = mean_squared_error(y_true, y_pred)

#calculate the R-squared
r2 = r2_score(y_true, y_pred)

#calculate the MAPE
mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100

#calculate the MAE
mae = mean_absolute_error(y_true, y_pred)

#residual plot with residual on the Y-axis and observed value (SalePrice)
residuals = y_true - y_pred

plt.figure(figsize=(10, 6))
plt.scatter(y_true, residuals, alpha=0.5)
plt.axhline(0, color='red', linestyle='--')

#calculate and plot the lowess smoothing line
lowess_results = lowess(residuals, y_true)
plt.plot(lowess_results[:, 0], lowess_results[:, 1], color='blue')

plt.xlabel('Observed Sale Price')
plt.ylabel('Residuals')
plt.title('Residual Plot with LOESS Line')
plt.grid(True)
plt.show()

#print the values
print(f'Sum Squared of Error (SSE): {sse}')
print(f'Mean Squared Error (MSE): {mse}')
print(f'R-squared : {r2}')
print(f'Mean Absolute Percentage Error (MAPE): {mape}')
print(f'Mean Absolute Error (MAE): {mae}')
